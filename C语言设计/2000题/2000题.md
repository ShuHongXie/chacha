int a[5] = {13,20,20,40,50};
int b[5] = {0,3,1,2,0};
int i,sum = 0;
for(i=0;i<5;i++) {
  sum = sum + a[b[i]];
}
printf("%d\n",sum);
// 第一轮 i=0, sum = 0 + a[0] = 13
// 第二轮 i=1, sum = 13 + a[3] = 13+40 = 53
// 第三轮 i=2, sum  = 13 + a[1] =  53 + 20 = 73
// 第四轮 i=3, sum = 73 + a[2] = 73+30 = 103
// 第五轮 i=4, sum = 103 + a[0] = 103 + 13 = 116
// i = 5
